async function searchModel() { const input = document.getElementById('query').value; const loader = document.getElementById('loader'); const modelViewer = document.getElementById('main-model'); if(!input) return; loader.style.display = 'block'; try { // 1. Шукаємо .glb файли у відкритих репозиторіях GitHub const response = await fetch(`https://api.github.com/search/code?q=${input}+extension:glb`); const data = await response.json(); if (data.items && data.items.length > 0) { // 2. Беремо посилання на перший знайдений файл const fileData = data.items[0]; const rawUrl = `https://raw.githubusercontent.com/${fileData.repository.full_name}/master/${fileData.path}`; // 3. Завантажуємо в плеєр modelViewer.src = rawUrl; console.log("Модель знайдена:", rawUrl); } else { alert("Нічого не знайдено. Спробуйте англійською, наприклад: 'Dragon' або 'Vase'"); } } catch (e) { alert("Помилка пошуку. Спробуйте пізніше."); } finally { loader.style.display = 'none'; } }
